````python
STUDIES = [
    {
        'title': 'Book 1',
        'type': 'pages',
        'total': 871,
        'done': 250,
    },
    {
        'title': 'Course 1',
        'type': 'classes',
        'total': 512,
        'done': 124,
    }
]

EXAM = {
    'name': 'Exam 1',
    'date': '31/12/2025',  # DD/MM/YYYY

}
````
